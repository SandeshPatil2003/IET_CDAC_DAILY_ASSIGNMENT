select s.salesman_name, c.customer_name, sa.amount from sales sa join salesman s on sa.salesman_id = s.salesman_id join customer c on sa.customer_id = c.customer_id;


2.

select e.ename, m.ename as manager_name from emp e join emp m on e.mgr = m.empno;

3.

select d., e. from dept d left join emp e on d.deptno = e.deptno;

4.

select c.customer_name, o.order_number, o.ship_date from customer c join orders o on c.customer_id = o.customer_id where year(o.ship_date) = 1987;

5.

select s.salesman_name, c.customer_name from salesman s join customer c on s.salesman_id = c.salesman_id;

6.

select s.salesman_name, c.customer_name from salesman s join customer c on s.salesman_id = c.salesman_id where s.comm is null or s.comm = 0;

7.

select s.salesman_name, s.sal, s.comm, sg.grade from salesman s join salgrade sg on s.sal between sg.losal and sg.hisal;

8.

select p.product_name, p.standard_price, p.minimum_price from product p where p.category = 'tennis';

9.

select p.product_name from product p where p.customer_id = 100 or p.customer_id = 103;


10.

select d.dname, e.sal from emp e join dept d on e.deptno = d.deptno where (e.sal, e.comm) in (select sal, comm from emp where loc = 'dallas');

11.

select distinct e.job from emp e join dept d on e.deptno = d.deptno where d.dname in ('research', 'accounting') order by e.job asc;
